inline bool IsPointerValid(void* ptr)
{
    if (!ptr)
        return false;
    MEMORY_BASIC_INFORMATION mbi;
    if (VirtualQuery(ptr,&mbi, sizeof(mbi))==0)
        return false;
    if (mbi.State!= MEM_COMMIT)
        return false;
    DWORD protect = mbi.Protect;
    if (protect& (PAGE_NOACCESS | PAGE_GUARD))
        return false;
    if (!(protect& (PAGE_READONLY | PAGE_READWRITE | PAGE_EXECUTE_READ | PAGE_EXECUTE_READWRITE)))
        return false;
    return true;
}
2.1 获取 cs2 所有字段的偏移
在 cs2 中，客户端的所有实体变量（netvar）都是通过 schema 系统进行上传的，所以我们可以通过 schema 接口获取所有字段的名字，变量类型和相对他结构体的偏移，这也解释了为什么获取 cs2 的实体名字和偏移总是很轻松可以获取。在本文中，我们将使用此工具进行获取：https://github.com/a2x/cs2-dumper/releases/download/0.1.2/cs2-dumper.exe启动游戏运行 exe，他就会生成一个 output 文件夹

GetBaseEntity（）
cpp
运行
uintptr_t GetBaseEntity(int index,uintptr_t client){
    auto entListBase=*reinterpret_cast<std::uintptr_t*>(client+cs2_dumper::offsets::client_dll::dwEntityList);
    if (entListBase==0){
        return 0;
    }
    auto entitylistbase=*reinterpret_cast<std::uintptr_t*>(entListBase+0x8*(index>>9)+16);
    if (entitylistbase==0){
        return 0;
    }
    return *reinterpret_cast<std::uintptr_t*>(entitylistbase+(0x78*(index & 0x1FF)));
}
GetBaseEntity（）这个函数返回一个实体的控制器，对应的是 client 里的 C_BaseEntity 类型，所以 C_BaseEntity 下的字段对这个控制器可用。想象一个超市在一个城市里（城市就是 client），整个超市就是 dwEntityList（存放所有实体），超市里有很多货架，每个货架最多能放 512 件商品（每个分区最多存 512 个实体，所以 index >> 9 等价于 index/512），然后再通过只保留低 9 位（index & 0x1FF）的方式算出商品在货架的哪个位置
GetBaseEntityFromHandle（）
GetBaseEntityFromHandle（）这个函数返回一个实体的 Pawn（基址 / 本体），对应的是 client 里的 C_CSPlayerPawn C_CSWeaponBase 类型，所以他继承 C_BaseEntity 和他本身这个类，多个字段对这个实体可用。与 GetBaseEntity（）不同的是他会传入一个实体的 uhandle（句柄），并通过 uhandle & 0x7FFF 的方式取低 15 位获取真实索引，其他的没有任何不同。
GetEyePos（）
cpp
运行
std::optional<Vector3> GetEyePos(uintptr_t addr) noexcept{
    auto Origin=reinterpret_cast<Vector3*>(addr+cs2_dumper::schemas::client_dll::C_BasePlayerPawn::m_vOldOrigin);
    auto ViewOffset= reinterpret_cast<Vector3*>(addr+cs2_dumper::schemas::client_dll::CBaseModelEntity::m_vecViewOffset);
    Vector3 LocalEye= *Origin+*ViewOffset;
    if(std::isinf(LocalEye.x)||std::isinf(LocalEye.y)||std::isinf(LocalEye.z))
        return std::nullopt;
    if (LocalEye.Length()<0.1f)
        return std::nullopt;
    return LocalEye;
}
GetEyePos（）函数：此函数通过 abs_Origin 玩家脚底的位置（世界坐标）加上 ViewOffset （从脚底到眼睛的偏移），得到敌人的眼球坐标（也叫摄像机坐标）
WorldToScreen（）
cpp
运行
bool WorldToScreen(Vector3& pWorldPos,Vector3& pScreenPos,float* pMatrixPtr,float pWinWidth,float pWinHeight)
{
    float matrix2[4][4];
    memcpy(matrix2,pMatrixPtr,16*sizeof(float));
    const float mX = pWinWidth/2;
    const float mY = pWinHeight/2;
    const float w =
    matrix2[3][0]*pWorldPos.x +
    matrix2[3][1]*pWorldPos.y +
    matrix2[3][2]*pWorldPos.z +
    matrix2[3][3];
    if (w<0.65f) return false;
    const float x =
    matrix2[0][0]*pWorldPos.x+
    matrix2[0][1]*pWorldPos.y+
    matrix2[0][2]*pWorldPos.z+
    matrix2[0][3];
    const float y =
    matrix2[1][0]*pWorldPos.x+
    matrix2[1][1]*pWorldPos.y+
    matrix2[1][2]*pWorldPos.z+
    matrix2[1][3];
    pScreenPos.x=(mX+mX*x/w);
    pScreenPos.y=(mY-mY*y/w);
    pScreenPos.z=0;
    return true;
}
WorldToScreen（）函数：这个函数就是把 3D 点投影到屏幕上，步骤很简单：拷贝传进来的 4×4 矩阵用矩阵第 4 行算出 w，如果 w < 0.65，点离相机太近或在背后，返回 false用矩阵第 1、2 行算出 x、y（还没除以 w 的裁剪坐标）做透视除法 x/w、y/w，再 × 半屏宽 / 高并加上屏幕中心，得出最终屏幕坐标 pScreenPos返回 true，表示这个点可见并已转换好。
